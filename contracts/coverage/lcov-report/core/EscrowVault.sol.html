// SPDX-License-Identifier: Apache-2.0
pragma solidity ^0.8.20;
 
import "@openzeppelin/contracts/access/Ownable.sol";
import "@openzeppelin/contracts/security/ReentrancyGuard.sol";
 
contract EscrowVault is Ownable, ReentrancyGuard {
    // Mapping of policyId => deposited premium amount
    mapping(uint256 => uint256) public premiums;
 
    // Authorized callers (e.g. KilimaShieldOracle, PolicyRegistry)
    mapping(address => bool) public authorizedCallers;
 
    event PremiumDeposited(uint256 indexed policyId, uint256 amount);
    event PayoutExecuted(uint256 indexed policyId, address indexed farmer, uint256 amount);
    event RefundExecuted(uint256 indexed policyId, address indexed farmer, uint256 amount);
    event CallerAuthorized(address caller, bool authorized);
 
    modifier onlyAuthorized() {
        Erequire(msg.sender == owner() || authorizedCallers[msg.sender], "Caller not authorized");
        _;
    }
 
    function setAuthorizedCaller(address caller, bool authorized) external EonlyOwner {
        authorizedCallers[caller] = authorized;
        emit CallerAuthorized(caller, authorized);
    }
 
    function depositPremium(uint256 policyId) external payable {
        Erequire(msg.value > 0, "Premium must be greater than 0");
        premiums[policyId] += msg.value;
        emit PremiumDeposited(policyId, msg.value);
    }
 
    function payout(
        uint256 policyId,
        address payable farmer,
        uint256 amount
    ) external EonlyAuthorized EnonReentrant {
        Erequire(farmer != address(0), "Invalid farmer address");
        Erequire(amount > 0, "Payout amount must be positive");
        Erequire(address(this).balance >= amount, "Insufficient contract balance");
 
        (bool success, ) = farmer.call{value: amount}("");
        Erequire(success, "Payout transfer failed");
 
        emit PayoutExecuted(policyId, farmer, amount);
    }
 
    function refundPremium(
        uint256 policyId,
        address payable farmer,
        uint256 amount
    ) external onlyAuthorized nonReentrant {
        require(farmer != address(0), "Invalid farmer address");
        require(amount > 0, "Refund amount must be positive");
        require(address(this).balance >= amount, "Insufficient contract balance");
 
        (bool success, ) = farmer.call{value: amount}("");
        require(success, "Refund transfer failed");
 
        emit RefundExecuted(policyId, farmer, amount);
    }
 
    // Function to receive Ether
    receive() external payable {}
}
